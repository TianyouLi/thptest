CMAKE_MINIMUM_REQUIRED ( VERSION 2.6)

PROJECT( thptest )

## set output directory
set ( CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/../lib )
set ( CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/../lib )
set ( CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/../bin )

file ( GLOB_RECURSE thptest_SRC
  "${CMAKE_SOURCE_DIR}/../src/thptest.cpp",
  "${CMAKE_SOURCE_DIR}/../src/page-info.c"
)
	
ADD_EXECUTABLE ( thptest  ${thptest_SRC} )

TARGET_LINK_LIBRARIES ( thptest
	
)

SET ( GCC_COMPILE_FLAGS "-D_FILE_OFFSET_BITS=64 -DFUSE_USE_VERSION=27 -Wall -O3  -std=c++2a -g")
SET ( GCC_LINK_FLAGS    "-lpthread -g")

SET( CMAKE_CXX_FLAGS  "${CMAKE_CXX_FLAGS} ${GCC_COMPILE_FLAGS}" )
SET( CMAKE_EXE_LINKER_FLAGS  "${CMAKE_EXE_LINKER_FLAGS} ${GCC_LINK_FLAGS}" )
